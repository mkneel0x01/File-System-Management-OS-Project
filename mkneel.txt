#include <stdio.h>
int main(){

    int p[20],bt[20],wt[20],tat[20],i,n;
    float wtavg,tatavg;

    printf("Enter the number of processes : ");
    scanf("%d",&n);
    printf(\n);

    for(i=0;i<n;i++){
        printf("Enter the brust time : ",i);
        scanf("%d",bt[i]);
    }
    wt[0] = wtavg = 0;
    tat [0] = tatavg = bt[0];

    for(i=0; i<n; i++){
        wt[i] = tat[i-1];
        tat[i] = bt[i] + wt[i];
        wtavg = wtavg + wt[i];
        tatavg = tatavg + tat[i];
    }

    printf("\nProcess\t\Brust Time\tWaiting Time\tTroundAround Time");

    for(i=0;i<n;i++){
        printf("\nP%d \t\t %d \t\t %d \t\t %d", i,bt[i],wt[i],tat[i]);
    }

    printf("\n");
    printf("Avg waiting time -----> %f \n", wtavg/n);
    printf("Avg TroundAround time -----> %f \n", tatavg/n);
    printf("\n");

    return 0;


}
